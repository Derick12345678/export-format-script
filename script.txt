param (
    [Parameter(Mandatory = $true)]
    [string]$InputFile
)

Import-Module ImportExcel

function Clean-ExcelValue {
    param (
        [AllowNull()]
        $Value
    )

    if ($null -eq $Value) {
        return $null
    }

    $text = $Value.ToString()

    # Remove hidden/control characters that can break Excel XML saving
    $text = $text -replace '[\x00-\x08\x0B\x0C\x0E-\x1F]', ''

    # Normalize line endings
    $text = $text -replace "`r`n", "`n"
    $text = $text -replace "`r", "`n"

    return $text
}

function Write-WorksheetManually {
    param (
        $ExcelPackage,
        [string]$WorksheetName,
        [array]$Rows
    )

    if ($null -eq $Rows -or $Rows.Count -eq 0) {
        Write-Host "No rows to write for worksheet: $WorksheetName"
        return
    }

    # Delete existing sheet with same name if it somehow exists
    $existingSheet = $ExcelPackage.Workbook.Worksheets[$WorksheetName]
    if ($null -ne $existingSheet) {
        $ExcelPackage.Workbook.Worksheets.Delete($WorksheetName)
    }

    # Create worksheet
    $worksheet = $ExcelPackage.Workbook.Worksheets.Add($WorksheetName)

    # Get headers from first row
    $headers = @($Rows[0].PSObject.Properties.Name)

    # Write headers
    for ($col = 0; $col -lt $headers.Count; $col++) {
        $worksheet.Cells[1, $col + 1].Value = $headers[$col]
        $worksheet.Cells[1, $col + 1].Style.Font.Bold = $true
    }

    # Write data rows
    $rowNumber = 2

    foreach ($row in $Rows) {
        for ($col = 0; $col -lt $headers.Count; $col++) {
            $header = $headers[$col]
            $value = $row.PSObject.Properties[$header].Value

            $worksheet.Cells[$rowNumber, $col + 1].Value = Clean-ExcelValue $value
        }

        $rowNumber++
    }

    # Freeze top row
    $worksheet.View.FreezePanes(2, 1)

    # Add autofilter
    $lastRow = $Rows.Count + 1
    $lastCol = $headers.Count

    if ($lastRow -ge 1 -and $lastCol -ge 1) {
        $worksheet.Cells[1, 1, $lastRow, $lastCol].AutoFilter = $true
    }

    # Basic width, safer than AutoSize
    for ($col = 1; $col -le $lastCol; $col++) {
        $worksheet.Column($col).Width = 20
    }
}

# If only a file name is given, assume it is in the same folder as this script
$scriptFolder = $PSScriptRoot

if (-not [System.IO.Path]::IsPathRooted($InputFile)) {
    $InputFile = Join-Path $scriptFolder $InputFile
}

if (-not (Test-Path $InputFile)) {
    Write-Error "File not found: $InputFile"
    exit 1
}

# Build output file path
$folder = Split-Path $InputFile
$fileNameWithoutExtension = [System.IO.Path]::GetFileNameWithoutExtension($InputFile)

$timestamp = Get-Date -Format "yyyy-MM-dd_HH-mm-ss"

# Final output goes beside the original Excel file
$outputFile = Join-Path $folder "$fileNameWithoutExtension`_formatted_$timestamp.xlsx"

# Temporary output is created locally first to avoid network-drive save issues
$tempOutputFile = Join-Path $env:TEMP "$fileNameWithoutExtension`_formatted_$timestamp.xlsx"

# Read Excel normally so headers are preserved
$data = Import-Excel -Path $InputFile

function Get-SafeWorksheetName {
    param (
        [string]$Name,
        [hashtable]$UsedNames
    )

    if ([string]::IsNullOrWhiteSpace($Name)) {
        $Name = "Sheet"
    }

    # Excel worksheet names cannot contain: \ / ? * [ ] :
    $safeName = $Name -replace '[\\\/\?\*\[\]\:]', '-'
    $safeName = $safeName.Trim("'").Trim()

    if ([string]::IsNullOrWhiteSpace($safeName)) {
        $safeName = "Sheet"
    }

    # Excel worksheet names have a 31-character limit
    if ($safeName.Length -gt 31) {
        $safeName = $safeName.Substring(0, 31)
    }

    $baseName = $safeName
    $counter = 2

    while ($UsedNames.ContainsKey($safeName)) {
        $suffix = " ($counter)"
        $maxBaseLength = 31 - $suffix.Length

        if ($baseName.Length -gt $maxBaseLength) {
            $safeName = $baseName.Substring(0, $maxBaseLength) + $suffix
        }
        else {
            $safeName = $baseName + $suffix
        }

        $counter++
    }

    $UsedNames[$safeName] = $true
    return $safeName
}

function Get-RegulationSheetName {
    param (
        [string]$Text
    )

    if ($Text -match '^\s*Regulation\s+(\d+)\b') {
        return "Regulation $($matches[1])"
    }

    return $null
}

function Test-IsRegulationStart {
    param (
        [string]$Text
    )

    return ($Text -match '^\s*Regulation\s+\d+\b')
}

function Test-IsRequirementOrObjectiveStart {
    param (
        [string]$Text
    )

    return ($Text -match '^\s*\d+\.\d+\b')
}

function Test-IsValidColumnC {
    param (
        [string]$Text
    )

    return (
        (Test-IsRegulationStart -Text $Text) -or
        (Test-IsRequirementOrObjectiveStart -Text $Text)
    )
}

# Parse/decompose Column AC into OE groups
function Get-OESections {
    param (
        [AllowNull()]
        $CellValue
    )

    $sections = @()

    if ($null -eq $CellValue) {
        return $sections
    }

    $text = Clean-ExcelValue $CellValue

    if ([string]::IsNullOrWhiteSpace($text)) {
        return $sections
    }

    # Normalize new lines
    $text = $text -replace "`r`n", "`n"
    $text = $text -replace "`r", "`n"

    # Split on separator lines made of underscores
    $chunks = $text -split '(?m)^\s*_+\s*$'

    foreach ($chunk in $chunks) {

        $lines = $chunk -split "`n" |
            ForEach-Object { $_.Trim() } |
            Where-Object { -not [string]::IsNullOrWhiteSpace($_) }

        # Normal case:
        # Line 1 = SDRL
        # Line 2 = WC#
        # Line 3 = CSC#
        # Line 4 = Document Section
        if ($lines.Count -ge 4) {
            $sections += [PSCustomObject]@{
                SDRL            = Clean-ExcelValue $lines[0]
                WCNumber        = Clean-ExcelValue $lines[1]
                CSCNumber       = Clean-ExcelValue $lines[2]
                DocumentSection = Clean-ExcelValue $lines[3]
            }
        }

        # Missing CSC# case:
        # Line 1 = SDRL
        # Line 2 = WC#
        # Line 3 = Document Section
        elseif ($lines.Count -eq 3) {
            $sections += [PSCustomObject]@{
                SDRL            = Clean-ExcelValue $lines[0]
                WCNumber        = Clean-ExcelValue $lines[1]
                CSCNumber       = "N/A"
                DocumentSection = Clean-ExcelValue $lines[2]
            }
        }
    }

    return $sections
}

function Merge-OESections {
    param (
        [AllowNull()]
        [array]$Sections
    )

    $mergedSections = @()

    if ($null -eq $Sections -or $Sections.Count -eq 0) {
        return $mergedSections
    }

    foreach ($section in $Sections) {

        $existingSection = $mergedSections | Where-Object {
            $_.SDRL -eq $section.SDRL -and
            $_.CSCNumber -eq $section.CSCNumber -and
            $_.WCNumber -eq $section.WCNumber
        } | Select-Object -First 1

        if ($null -ne $existingSection) {

            $currentDocumentSection = $section.DocumentSection

            # Avoid repeating the word "Section"
            if ($existingSection.DocumentSection -match '^\s*Section\b' -and $currentDocumentSection -match '^\s*Section\s+(.+)$') {
                $currentDocumentSection = $matches[1]
            }

            $existingSection.DocumentSection = Clean-ExcelValue "$($existingSection.DocumentSection), $currentDocumentSection"
        }
        else {
            $mergedSections += [PSCustomObject]@{
                SDRL            = Clean-ExcelValue $section.SDRL
                CSCNumber       = Clean-ExcelValue $section.CSCNumber
                WCNumber        = Clean-ExcelValue $section.WCNumber
                DocumentSection = Clean-ExcelValue $section.DocumentSection
            }
        }
    }

    return $mergedSections
}

function Convert-ToFormattedRow {
    param (
        $OriginalRow,
        [int]$MaxOECount
    )

    $properties = @($OriginalRow.PSObject.Properties)

    # Original Column C = index 2
    $originalColumnC = $null

    if ($properties.Count -gt 2) {
        $originalColumnC = $properties[2].Value
    }

    $columnCText = ""

    if ($null -ne $originalColumnC) {
        $columnCText = (Clean-ExcelValue $originalColumnC).Trim()
    }

    $finalColumnA = $null
    $finalColumnB = $null
    $finalColumnC = $originalColumnC

    # Case 1: Column C starts with decimal/double, like 4.1 or 4.2
    if ($columnCText -match '^\s*(\d+\.\d+)\s*(.*)$') {

        $numberAtStart = $matches[1]
        $textAfterNumber = $matches[2].Trim()

        $finalColumnA = $numberAtStart

        $decimalPart = [int]($numberAtStart.Split(".")[1])

        if ($decimalPart -eq 1) {
            $finalColumnB = "Functional Objective"
        }
        else {
            $finalColumnB = "Performance Requirements"
        }

        $finalColumnC = $textAfterNumber
    }

    # Case 2: Column C starts with Regulation X
    elseif ($columnCText -match '^\s*Regulation\s+(\d+)\b\s*(.*)$') {

        $regulationNumber = $matches[1]
        $textAfterNumber = $matches[2].Trim()

        $finalColumnA = $regulationNumber
        $finalColumnB = $textAfterNumber
        $finalColumnC = $originalColumnC
    }

    $newRow = [ordered]@{}

    # New Columns A, B, C
    $newRow["Unique Identifier (Ruleset Number)"] = Clean-ExcelValue $finalColumnA
    $newRow["Regulation"] = Clean-ExcelValue $finalColumnB
    $newRow["Requirement Text"] = Clean-ExcelValue $finalColumnC

    # Copy original Columns D through O exactly
    # D = index 3
    # O = index 14
    for ($i = 3; $i -le 14; $i++) {

        if ($i -lt $properties.Count) {

            $property = $properties[$i]

            if ($null -ne $property -and -not [string]::IsNullOrWhiteSpace($property.Name)) {
                $newRow[$property.Name] = Clean-ExcelValue $property.Value
            }
        }
    }

    # Decompose original Column AC and insert OE columns after original Column O
    # AC = index 28
    $columnAC = $null

    if ($properties.Count -gt 28) {
        $columnAC = $properties[28].Value
    }

    $oeSections = @(Get-OESections -CellValue $columnAC)
    $mergedOESections = @(Merge-OESections -Sections $oeSections)

    for ($oeIndex = 1; $oeIndex -le $MaxOECount; $oeIndex++) {

        $arrayIndex = $oeIndex - 1

        if ($arrayIndex -lt $mergedOESections.Count) {
            $section = $mergedOESections[$arrayIndex]

            $newRow["OE $oeIndex (Add SDRL, if applicable)"] = Clean-ExcelValue $section.SDRL
            $newRow["OE $oeIndex CSC# (CSC#/T26#)"] = Clean-ExcelValue $section.CSCNumber
            $newRow["OE $oeIndex LMCa WC #"] = Clean-ExcelValue $section.WCNumber
            $newRow["OE $oeIndex Document Section"] = Clean-ExcelValue $section.DocumentSection
        }
        else {
            $newRow["OE $oeIndex (Add SDRL, if applicable)"] = "N/A"
            $newRow["OE $oeIndex CSC# (CSC#/T26#)"] = "N/A"
            $newRow["OE $oeIndex LMCa WC #"] = "N/A"
            $newRow["OE $oeIndex Document Section"] = "N/A"
        }
    }

    # Continue copying original Columns P through AB exactly
    # P = index 15
    # AB = index 27
    # Do NOT copy AC or anything after AC
    for ($i = 15; $i -le 27; $i++) {

        if ($i -lt $properties.Count) {

            $property = $properties[$i]

            if ($null -ne $property -and -not [string]::IsNullOrWhiteSpace($property.Name)) {
                $newRow[$property.Name] = Clean-ExcelValue $property.Value
            }
        }
    }

    return [PSCustomObject]$newRow
}

# STEP 1:
# Filter out rows where Column C is not either:
# - Regulation X
# - X.Y
$validRows = @()

foreach ($rowItem in $data) {

    $properties = @($rowItem.PSObject.Properties)

    $columnC = $null

    if ($properties.Count -gt 2) {
        $columnC = $properties[2].Value
    }

    $columnCText = ""

    if ($null -ne $columnC) {
        $columnCText = (Clean-ExcelValue $columnC).Trim()
    }

    if (Test-IsValidColumnC -Text $columnCText) {
        $validRows += $rowItem
    }
}

# STEP 2:
# Group rows by regulation.
# A new regulation/sheet starts only when Column C starts with "Regulation X"
$regulations = @()
$currentRegulation = $null

foreach ($rowItem in $validRows) {

    $properties = @($rowItem.PSObject.Properties)

    $columnC = $null

    if ($properties.Count -gt 2) {
        $columnC = $properties[2].Value
    }

    $columnCText = ""

    if ($null -ne $columnC) {
        $columnCText = (Clean-ExcelValue $columnC).Trim()
    }

    if (Test-IsRegulationStart -Text $columnCText) {

        $sheetName = Get-RegulationSheetName -Text $columnCText

        $currentRegulation = [PSCustomObject]@{
            OriginalName = $columnCText
            SheetName    = $sheetName
            Rows         = @()
        }

        $regulations += $currentRegulation
    }

    # Ignore rows before the first regulation
    if ($null -ne $currentRegulation) {
        $currentRegulation.Rows += $rowItem
    }
}

if ($regulations.Count -eq 0) {
    Write-Error "No regulation rows found. No output file was created."
    exit 1
}

# Remove existing files if needed
if (Test-Path $outputFile) {
    Remove-Item $outputFile -Force
}

if (Test-Path $tempOutputFile) {
    Remove-Item $tempOutputFile -Force
}

# STEP 3:
# Create one Excel package locally, add all regulation sheets manually, then save once at the end
$usedSheetNames = @{}

$excelPackage = Open-ExcelPackage -Path $tempOutputFile -Create

try {
    foreach ($regulation in $regulations) {

        # Find max OE count for this regulation only
        $maxOECount = 0

        foreach ($rowItem in $regulation.Rows) {

            $properties = @($rowItem.PSObject.Properties)

            # Column AC = index 28
            $columnAC = $null

            if ($properties.Count -gt 28) {
                $columnAC = $properties[28].Value
            }

            $oeSections = @(Get-OESections -CellValue $columnAC)
            $mergedOESections = @(Merge-OESections -Sections $oeSections)

            if ($mergedOESections.Count -gt $maxOECount) {
                $maxOECount = $mergedOESections.Count
            }
        }

        $worksheetName = Get-SafeWorksheetName -Name $regulation.SheetName -UsedNames $usedSheetNames

        Write-Host "Writing worksheet manually: $worksheetName"
        Write-Host "Regulation '$($regulation.OriginalName)' max OE count: $maxOECount"

        $formattedRows = @()

        foreach ($rowItem in $regulation.Rows) {
            $formattedRows += Convert-ToFormattedRow -OriginalRow $rowItem -MaxOECount $maxOECount
        }

        Write-WorksheetManually `
            -ExcelPackage $excelPackage `
            -WorksheetName $worksheetName `
            -Rows $formattedRows
    }
}
finally {
    Close-ExcelPackage -ExcelPackage $excelPackage
}

# Copy completed workbook from local temp folder back to the original folder
Copy-Item -Path $tempOutputFile -Destination $outputFile -Force

# Clean up local temp output file
if (Test-Path $tempOutputFile) {
    Remove-Item $tempOutputFile -Force
}

Write-Host "Formatted Excel file created:"
Write-Host $outputFile
